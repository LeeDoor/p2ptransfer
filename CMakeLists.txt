project(p2ptransfer)
set(CMAKE_EXPORT_COMPILE_COMMANDS 1)
find_package(Boost 1.88 REQUIRED)

add_executable(server server.cpp)

target_include_directories(server PUBLIC ${Boost_INCLUDE_DIRS})
